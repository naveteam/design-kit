# Dialog

O componente `Dialog` é reponsável por apresentar uma extensa variedade de conteúdos, seja na de transmissão de informações ao usuário, ou até mesmo na execução de 'actions'.

export const DialogCloseIcon = () => {
  const [isOpen, setIsOpen] = useState(false)
  return (
    <>
      <Button width={'150px'} onClick={() => setIsOpen(true)}>Open Modal</Button >
      <Dialog
        open={isOpen}
        onClose={setIsOpen}
        withCloseIcon
        title="Título"
        description="Lorem ipsum dolor sit amet, consectetur adipiscing elit."
        actionButton={{
          label: 'Adicionar'
        }}
      />
    </>
  )
}
