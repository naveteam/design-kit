import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs/blocks'

import { Avatar, SaturnProvider } from '../components'

<Meta title='Data Display/Avatar' component={Avatar} />

export const AvatarControl = args => (
  <SaturnProvider>
    <Avatar {...args} />
  </SaturnProvider>
)

# Avatar

O componente de `Avatar` é utilizado para mostrar o avatar de um usuário.

<Story
  name='Base'
  args={{
    avatar: 'https://source.unsplash.com/random/400x400'
  }}
  parameters={{
    design: {
      type: 'figma',
      url: 'https://www.figma.com/file/O3bKxIcsj2rc1FNIRclJyT/Saturn-System?node-id=915%3A0'
    }
  }}
>
  {AvatarControl.bind()}
</Story>

# Name

Caso não seja passado um atributo `avatar` é possível passar um do nome do avatar, para que sejam mostradas as suas iniciais, através da propriedade `name`.

<Canvas>
  <Story
    name='Inicial'
    parameters={{
      design: {
        type: 'figma',
        url: 'https://www.figma.com/file/O3bKxIcsj2rc1FNIRclJyT/Saturn-System?node-id=915%3A0'
      }
    }}
  >
    <SaturnProvider>
      <Avatar name='Lorem Ipsum' color='#ffffff' />
      <Avatar name='Fulano de Tal' color='#ffffff' size='large' />
    </SaturnProvider>
  </Story>
</Canvas>

# Default

Caso não haja as propriedades `avatar` e `name` será mostrado um ícone padrão para usuário e para empresa.

<Canvas>
  <Story
    name='Default'

</Canvas>
